### 2. 音频参数
CHUNK = 1024 帧数
FORMAT = pyaudio.pcInt16 模拟信号转数字信号，转换后用多长的字符存储
CHANNELS = 2 单声道或左右双声道
RATE = 44100 音频采样频率，44.1kHZ是CD音质

### 3. 录音存成音频SOP

1. 开启流
2. 流转存文件
3. 流处理
4. 循环read
5. 关闭并释放资源

```
import pyaudio
import wave
from tqdm import tqdm

def record_audio(wave_out_path,record_second):

    # 音频设备采样参数
    CHUNK = 1024
    FORMAT = pyaudio.paInt32
    CHANNELS = 1
    RATE = 44100

    # 初始化变量，确保在finally中可访问
    p = None
    stream = None
    wf = None

    try:
        # 初始化PyAudio
        p = pyaudio.PyAudio()

        # 创建流
        stream = p.open(format=FORMAT,
                        channels=CHANNELS,
                        rate=RATE,
                        input=True,
                        frames_per_buffer=CHUNK)

        # 创建文件，写入文件头
        wf = wave.open(wave_out_path, 'wb')
        wf.setnchannels(CHANNELS)
        wf.setsampwidth(p.get_sample_size(FORMAT))
        wf.setframerate(RATE)

        print("* recording")

        # 循环处理读取数据
        for i in tqdm(range(0, int(RATE / CHUNK * record_second))):
            data = stream.read(CHUNK)
            wf.writeframes(data)

        print("* done recording")

    except Exception as e:
        print(f"录制过程中发生错误: {e}")

    finally:        
        # 关闭流（如果已创建）
        if stream is not None:
            stream.stop_stream()
            stream.close()

        # 终止PyAudio（如果已初始化）
        if p is not None:
            p.terminate()

        # 关闭文件（如果已打开）
        if wf is not None:
            wf.close()

        print("资源清理完成")

record_audio("output2.wav",record_second=4)
```
